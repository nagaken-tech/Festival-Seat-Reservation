  <script>

    const successSound = new Audio('https://github.com/nagaken-tech/fun-festival-sounds/raw/refs/heads/main/Firefly_audio_A_short_but_elegant_boarding_gate_chime_for_a_prem_variation2.wav'); 

    const errorSound = new Audio('https://github.com/nagaken-tech/fun-festival-sounds/raw/refs/heads/main/%E8%AD%A6%E5%91%8A%E9%9F%B31.mp3');

    const scanSound = new Audio('https://github.com/nagaken-tech/fun-festival-sounds/raw/refs/heads/main/%E6%B1%BA%E5%AE%9A%E3%83%9B%E3%82%99%E3%82%BF%E3%83%B3%E3%82%92%E6%8A%BC%E3%81%9952.mp3');   


    // グローバル変数で現在の学籍番号を保持

    var currentStudentId = '';

    var isReservationConfirmed = false; // 予約が確認された状態か否か

    var ticketNotesDiv = document.getElementById('ticket-notes');

    var studentIdInput = document.getElementById('studentId'); // 学籍番号入力フィールドを取得


    // バーコードリーダーからの入力を監視し、8桁になったら自動で検索を実行

    studentIdInput.addEventListener('input', function() {

        if (this.value.length === 8) {

            scanSound.play(); // スキャン完了音を再生

            searchReservation();

        }

    });


    // エンターキーが押されたときに検索を実行するイベントリスナーを追加

    studentIdInput.addEventListener('keypress', function(event) {

        if (event.key === 'Enter') {

            event.preventDefault(); 

            searchReservation();

        }

    });


    function searchReservation() {

      if (document.getElementById('overlay').style.display === 'flex') return;

      

      currentStudentId = studentIdInput.value.trim();

      if (!currentStudentId) {

        alert('学籍番号を入力してください。');

        return;

      }


      var overlay = document.getElementById('overlay');

      overlay.textContent = '予約情報検索中...';

      overlay.className = ''; 

      overlay.style.display = 'flex';


      google.script.run

        .withSuccessHandler(function(response) {

          overlay.style.display = 'none';


          document.getElementById('reservation-details').innerHTML = response.ticketHtml || '';

          document.getElementById('info-display').style.display = 'block'; 

          document.getElementById('controls').style.display = 'none'; 

          var buttonsDiv = document.querySelector('#info-display .buttons');

          buttonsDiv.innerHTML = '';

          ticketNotesDiv.style.display = 'none';


          if (response.status === 'success') {

            successSound.play(); // 成功音を再生

            isReservationConfirmed = true;

            buttonsDiv.innerHTML = `

              <button onclick="confirmAndPrintTicket()">印刷・発券</button>

              <button class="cancel-button" onclick="resetPrinterPage()">キャンセル</button>

            `;

            ticketNotesDiv.style.display = 'block';

          } else if (response.status === 'issued') {

            errorSound.play(); // 失敗音を再生

            isReservationConfirmed = false;

            buttonsDiv.innerHTML = `

              <p style="color: red; font-weight: bold;">${response.message}</p>

              <button class="cancel-button" onclick="resetPrinterPage()">閉じる</button>

            `;

          } else { 

            errorSound.play(); // 失敗音を再生

            isReservationConfirmed = false;

            buttonsDiv.innerHTML = `

              <p style="color: red; font-weight: bold;">${response.message}</p>

              <button class="cancel-button" onclick="resetPrinterPage()">閉じる</button>

            `;

          }

        })

        .withFailureHandler(function(error) {

          errorSound.play(); // サーバーエラー時も失敗音を再生

          console.error("検索エラー:", error);

          overlay.textContent = 'エラー: ' + error.message;

          overlay.classList.add('error');

          isReservationConfirmed = false;

          setTimeout(function() {

              overlay.style.display = 'none';

              overlay.classList.remove('error');

              resetPrinterPage();

          }, 5000);

        })

        .searchReservationInfo(currentStudentId);

    }


    function confirmAndPrintTicket() {

      if (!isReservationConfirmed) {

          alert('不正な操作です。再度検索からやり直してください。');

          resetPrinterPage();

          return;

      }


      var overlay = document.getElementById('overlay');

      overlay.textContent = '発券処理中...';

      overlay.className = ''; 

      overlay.style.display = 'flex';


      google.script.run

        .withSuccessHandler(function(response) {

          if (response.status === 'success') {

            overlay.textContent = '発券済みマーク完了！\n印刷ダイアログを開きます...';

            overlay.classList.add('success'); 


            var originalOnAfterPrint = window.onafterprint;

            window.onafterprint = function() {

                window.onafterprint = originalOnAfterPrint;

                setTimeout(function() {

                    overlay.style.display = 'none';

                    overlay.classList.remove('success');

                    resetPrinterPage();

                }, 100); 

            };

            window.print();


          } else {

            errorSound.play();

            overlay.textContent = 'エラー: ' + response.message;

            overlay.classList.add('error');

            setTimeout(function() {

                overlay.style.display = 'none';

                overlay.classList.remove('error');

            }, 5000);

          }

        })

        .withFailureHandler(function(error) {

          errorSound.play();

          console.error("発券済みマークエラー:", error);

          overlay.textContent = 'エラー: ' + error.message;

          overlay.classList.add('error');

          setTimeout(function() {

              overlay.style.display = 'none';

              overlay.classList.remove('error');

          }, 5000);

        })

        .markTicketAsIssued(currentStudentId); 

    }


    function resetPrinterPage() {

      studentIdInput.value = '';

      document.getElementById('info-display').style.display = 'none'; 

      document.getElementById('controls').style.display = 'flex'; 

      studentIdInput.focus();

      currentStudentId = ''; 

      isReservationConfirmed = false; 

      document.getElementById('overlay').style.display = 'none'; 

      document.querySelector('#info-display .buttons').innerHTML = ''; 

      document.getElementById('reservation-details').innerHTML = ''; 

      ticketNotesDiv.style.display = 'none';

    }


    document.addEventListener('DOMContentLoaded', function() {

        document.getElementById('overlay').style.display = 'none'; 

        resetPrinterPage(); 

    });


    // ページ全体でキーボードの入力を監視し、Enter/Spaceキーでの操作を可能にする

    document.addEventListener('keydown', function(event) {

        if (event.key !== 'Enter' && event.key !== ' ' && event.code !== 'Space') return;


        const infoDisplay = document.getElementById('info-display');

        if (infoDisplay.style.display === 'block') {

            event.preventDefault();


            if (event.key === 'Enter') {

                const confirmButton = infoDisplay.querySelector('button[onclick="confirmAndPrintTicket()"]');

                if (confirmButton) {

                    confirmButton.click();

                    return;

                }

                const closeButton = infoDisplay.querySelector('.cancel-button');

                if (closeButton) {

                    closeButton.click();

                }

            } 

            else if (event.key === ' ' || event.code === 'Space') {

                const confirmButton = infoDisplay.querySelector('button[onclick="confirmAndPrintTicket()"]');

                const cancelButton = infoDisplay.querySelector('.cancel-button');

                if (confirmButton && cancelButton) {

                    cancelButton.click();

                }

            }

        }

    });
