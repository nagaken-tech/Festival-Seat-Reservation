  <script>

    // どの列がどのグループか (Code.gs の定義と一致させる)

    var GROUP_COLUMNS_MAP = {

      'GROUP_1': ['D', 'E', 'F', 'G', 'H', 'I', 'J', 'K'],

      'GROUP_2': ['L', 'M', 'N', 'O', 'P', 'Q', 'R'],

      'GROUP_3': ['S', 'T', 'U', 'V', 'W', 'X', 'Y']

    };


    // 起動時に空席リストを取得して描画する

    document.addEventListener('DOMContentLoaded', function() {

      google.script.run

        .withSuccessHandler(function(response) {

          // --- ▼▼▼ ここからが修正箇所です ▼▼▼ ---

          var loadingDiv = document.getElementById('loading');

          

          // response が存在し、かつ status が 'success' の場合のみ成功

          if (response && response.status === 'success') {

            loadingDiv.style.display = 'none'; // 成功時のみ「読み込み中」を非表示

            drawTickets(response.ticketHtmls); // HTMLのリストを受け取る

          

          } else {

            // サーバー側でエラーが起きた場合 (status: 'error' や、予期せぬ応答)

            var errorMsg = 'エラー: ';

            if (response && response.message) {

              errorMsg += response.message;

            } else if (response) {

              errorMsg += 'サーバーから予期せぬ応答がありました。';

            } else {

              errorMsg += 'サーバーから応答がありません。';

            }

            

            loadingDiv.textContent = errorMsg;

            loadingDiv.style.color = 'red';

            loadingDiv.style.display = 'block'; // ★エラーメッセージを確実に表示する

          }

          // --- ▲▲▲ ここまでが修正箇所です ▲▲▲ ---

        })

        .withFailureHandler(function(error) {

          // 通信自体が失敗した場合 (ここの処理は元のまま)

          document.getElementById('loading').textContent = 'サーバーエラー: ' + error.message;

          document.getElementById('loading').style.color = 'red';

        })

        .getAvailableTicketHtmlList(); // ★新しく作ったバックエンド関数を呼ぶ

    });


    // チケットのHTMLをコンテナに追加する

    function drawTickets(ticketHtmlList) {

      var container = document.getElementById('ticket-container');

      container.innerHTML = '';

      

      if (!ticketHtmlList || ticketHtmlList.length === 0) {

          var loadingDiv = document.getElementById('loading');

          loadingDiv.textContent = '空席はありません。（満席です）';

          loadingDiv.style.display = 'block';

          loadingDiv.style.color = '#333'; // エラーではないので黒色に

          return;

      }

      

      // 複製・譲渡不可の注意書きを追加

      var notes = `

        <div class="ticket-notes">

          複製・第三者譲渡不可<br>聖学院生のみ有効

        </div>

      `;


      ticketHtmlList.forEach(function(html) {

        var ticket = document.createElement('div');

        ticket.className = 'ticket';

        

        // `generateTicketHtml` が生成したHTML + 注意書き を挿入

        ticket.innerHTML = html + notes; 

        

        container.appendChild(ticket);

      });

    }


    // 座席番号 (例: 'F5') からグループ名 (例: 'GROUP 1') を返す

    function getGroupName(seatId) {

      var col = seatId.charAt(0).toUpperCase();

      if (GROUP_COLUMNS_MAP['GROUP_1'].indexOf(col) !== -1) return 'GROUP 1';

      if (GROUP_COLUMNS_MAP['GROUP_2'].indexOf(col) !== -1) return 'GROUP 2';

      if (GROUP_COLUMNS_MAP['GROUP_3'].indexOf(col) !== -1) return 'GROUP 3';

      return 'その他';

    }
