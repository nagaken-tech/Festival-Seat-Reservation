    var seatMap = {

      'D': [6, 17], 'E': [4, 17], 'F': [4, 17],

      'G': [3, 17], 'H': [2, 17], 'I': [1, 17], 'J': [1, 17], 'K': [1, 17],

      'L': [2, 17], 'M': [2, 17], 'N': [2, 17], 'O': [2, 17], 'P': [2, 17],

      'Q': [2, 17], 'R': [2, 17], 'S': [2, 17], 'T': [2, 17], 'U': [2, 17],

      'V': [2, 17], 'W': [2, 17], 'X': [2, 17], 'Y': [2, 17]

    };


    var selectedSeats = [];

    var reservedSeats = [];

    var reservedIds = [];

    var specialId = ‘*********’;

    var specialSeat = 'E4';


    document.getElementById('overlay').style.whiteSpace = 'pre-line';


    const currentYear = new Date().getFullYear();

    const gradeToAdmissionYearMap = {

        '中１': currentYear - 0,

        '中２': currentYear - 1,

        '中３': currentYear - 2,

        '高１': currentYear - 3,

        '高２': currentYear - 4,

        '高３': currentYear - 5

    };


    function drawSeats() {

      var container = document.getElementById('seat-container');

      container.innerHTML = '';


      var rowLabelColumn = document.createElement('div');

      rowLabelColumn.className = 'row-label-column';

      rowLabelColumn.appendChild(document.createElement('div')).className = 'row-label-empty';

      for (var row = 1; row <= 17; row++) {

        var label = document.createElement('div');

        label.className = 'row-label';

        label.textContent = row;

        rowLabelColumn.appendChild(label);

      }

      container.appendChild(rowLabelColumn);


      var noticeDiv = document.createElement('div');

      noticeDiv.id = 'hidden-cols-notice';

      noticeDiv.innerHTML = '出演者等待機席のため<br>A～C列は<br>発券いたしません';

      container.appendChild(noticeDiv);


      var cols = Object.keys(seatMap).sort();

      var currentStudentId = document.getElementById('id').value.trim();

      var isSpecialUser = (currentStudentId === specialId);


      for (var i = 0; i < cols.length; i++) {

        var col = cols[i];

        var colData = seatMap[col];

        var start = colData[0];

        var end = colData[1];


        var colDiv = document.createElement('div');

        colDiv.className = 'column';


        var label = document.createElement('div');

        label.className = 'col-label';

        label.textContent = col;

        colDiv.appendChild(label);


        for (var row = 1; row <= 17; row++) {

          var cell = document.createElement('div');

          var id = col + row;


          if (col === 'A' || col === 'B' || col === 'C') {

              cell.className = 'empty';

          } else if (row < start || row > end) {

            cell.className = 'empty';

          } else {

            cell.textContent = id;

            if (reservedSeats.indexOf(id) !== -1) {

              cell.className = 'reserved';

            } else if (isSpecialUser && id !== specialSeat) {

              cell.className = 'reserved';

            } else {

              cell.className = 'seat';

              cell.onclick = (function(c_cell, c_id) {

                return function() { toggleSeat(c_cell, c_id); };

              })(cell, id);

              if (selectedSeats.indexOf(id) !== -1) {

                cell.classList.add('selected');

              }

            }

          }

          colDiv.appendChild(cell);


          if (row === 5 || row === 13) {

            var aisle = document.createElement('div');

            aisle.className = 'aisle';

            colDiv.appendChild(aisle);

          }

        }

        container.appendChild(colDiv);


        if (col === 'K') {

            var verticalAisle = document.createElement('div');

            verticalAisle.className = 'vertical-aisle';

            container.appendChild(verticalAisle);

        }

      }

    }


    function toggleSeat(el, id) {

      if (id.charAt(0) === 'A') return;


      var countInput = document.getElementById('count');

      var max = parseInt(countInput.value);


      var currentStudentId = document.getElementById('id').value.trim();

      var isSpecialUser = (currentStudentId === specialId);


      if (isSpecialUser && id !== specialSeat) return;

      if (isSpecialUser && selectedSeats.length > 0 && selectedSeats[0] === specialSeat && id !== specialSeat) {

          alert('この学籍番号では座席 ' + specialSeat + ' のみ予約可能です。');

          return;

      }


      if (isNaN(max) || max < 1) {

          alert('有効な人数を入力してください（1以上）。');

          return;

      }


      if (reservedSeats.indexOf(id) !== -1) return;


      if (selectedSeats.indexOf(id) !== -1) {

        selectedSeats = selectedSeats.filter(function(s) { return s !== id; });

        el.classList.remove('selected');

      } else if (selectedSeats.length < max) {

        selectedSeats.push(id);

        el.classList.add('selected');

      } else {

        alert('選択した座席の数（' + max + '）以上は選択できません。');

      }

    }


    function submit() {

      var count = parseInt(document.getElementById('count').value);

      var studentId = document.getElementById('id').value.trim();

      var studentName = document.getElementById('name').value.trim();

      var selectedGrade = document.getElementById('grade').value;


      if (selectedSeats.length === 0) { alert('座席を選択してください。'); return; }

      if (selectedSeats.length !== count) { alert('選択した座席の数と入力された人数が一致していません。'); return; }

      if (!studentId || !studentName) { alert('学籍番号と名前を入力してください。'); return; }


      const idPattern = /^b(\d{4})\d{3}$/;

      const match = studentId.match(idPattern);

      if (!match) { alert('学籍番号の形式が正しくありません。（例: bYYYYXXX）'); return; }


      const admissionYear = parseInt(match[1]);

      const expectedAdmissionYear = gradeToAdmissionYearMap[selectedGrade];

      if (admissionYear !== expectedAdmissionYear) {

          alert('学年と学籍番号の入学年が一致しません。\n選択された学年: ' + selectedGrade + ' (入学年: ' + expectedAdmissionYear + ')\n入力された学籍番号の入学年: ' + admissionYear);

          return;

      }


      var overlay = document.getElementById('overlay');

      overlay.textContent = '予約処理中...';

      overlay.style.display = 'flex';

      overlay.className = '';


      var data = {

        grade: selectedGrade,

        class: document.getElementById('class').value,

        id: studentId,

        name: studentName,

        count: count,

        seats: selectedSeats

      };


      if (google && google.script && google.script.run) {

        google.script.run

          .withSuccessHandler(function(msg) {

            overlay.textContent = msg || '予約が完了しました';

            overlay.classList.add('success');

            google.script.run.withSuccessHandler(function(response) {

                reservedSeats = response.reservedSeats || [];

                reservedIds = response.reservedIds || [];

                selectedSeats = [];

                drawSeats();

            }).getReservedSeats();

            setTimeout(function(){ overlay.style.display='none'; overlay.classList.remove('success'); }, 3500);

          })

          .withFailureHandler(function(err) {

            overlay.textContent = 'エラー: ' + (err && err.message ? err.message : '不明なエラー');

            overlay.classList.add('error');

            google.script.run.withSuccessHandler(function(response) {

                reservedSeats = response.reservedSeats || [];

                reservedIds = response.reservedIds || [];

                selectedSeats = [];

                drawSeats();

            }).getReservedSeats();

            setTimeout(function(){ overlay.style.display='none'; overlay.classList.remove('error'); }, 3500);

          })

          .submitReservation(data);

      } else {

        setTimeout(function(){ overlay.style.display='none'; alert('（開発モード）送信シミュレーション完了'); }, 800);

      }

    }


    document.getElementById('id').addEventListener('input', function() {

        var currentStudentId = this.value.trim();

        var countInput = document.getElementById('count');


        if (currentStudentId === specialId) {

            if (!countInput.dataset.originalValue) countInput.dataset.originalValue = countInput.value;

            countInput.value = 1;

            countInput.setAttribute('disabled', 'true');

            selectedSeats = [specialSeat];

            drawSeats();

        } else {

            countInput.removeAttribute('disabled');

            if (countInput.dataset.originalValue) { countInput.value = countInput.dataset.originalValue; delete countInput.dataset.originalValue; }

            if (selectedSeats.length === 1 && selectedSeats[0] === specialSeat) selectedSeats = [];

            drawSeats();

        }

    });


    document.addEventListener('DOMContentLoaded', function() {

        var overlay = document.getElementById('overlay');

        overlay.textContent = '受付状況を確認中...';

        overlay.style.display = 'flex';

        if (google && google.script && google.script.run) {

            google.script.run

              .withSuccessHandler(function(status) {

                if (status && status.isOpen) {

                  overlay.textContent = '座席データを読み込み中...';

                  google.script.run.withSuccessHandler(function(response){

                    reservedSeats = response.reservedSeats || [];

                    reservedIds = response.reservedIds || [];

                    drawSeats();

                    overlay.style.display = 'none';

                  }).withFailureHandler(function(e){ overlay.textContent='読み込みエラー'; console.error(e); }).getReservedSeats();

                } else {

                  document.getElementById('controls').style.display='none';

                  document.getElementById('seat-area').style.display='none';

                  overlay.textContent = (status && status.message) ? status.message : '受付は終了しました';

                  overlay.classList.add('error');

                }

              })

              .withFailureHandler(function(e){ overlay.textContent='読み込みエラー'; console.error(e); })

              .getDeadlineStatus();

        } else {

          reservedSeats = [];

          reservedIds = [];

          drawSeats();

          overlay.style.display='none';

        }

    });
